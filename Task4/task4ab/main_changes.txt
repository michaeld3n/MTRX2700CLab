MAIN.C CHANGES - TASK 4 (a) and (b)
Magnetometer only - basic heading
=====================================


--------------------------------------------------------------
1. USER CODE BEGIN Includes
--------------------------------------------------------------

  /* USER CODE BEGIN Includes */
  #include "compass.h"
  #include <stdio.h>
  #include <string.h>
  /* USER CODE END Includes */


--------------------------------------------------------------
2. USER CODE BEGIN 2  (after all MX_ init calls)
--------------------------------------------------------------

  /* USER CODE BEGIN 2 */
  compass_init();
  /* USER CODE END 2 */


--------------------------------------------------------------
3. While loop
--------------------------------------------------------------

  /* USER CODE BEGIN WHILE */
  char buf[80];
  while (1)
  {
    /* USER CODE END WHILE */

    /* USER CODE BEGIN 3 */
    compass_read();
    CompassData *d = compass_get_data();

    int h_int  = (int)d->heading;
    int h_frac = (int)((d->heading - h_int) * 10);

    sprintf(buf, "Heading: %d.%d deg | X:%d Y:%d Z:%d | T:%u ms\r\n",
            h_int, h_frac,
            d->raw_x, d->raw_y, d->raw_z,
            (unsigned int)d->timestamp);

    HAL_UART_Transmit(&huart1, (uint8_t*)buf, strlen(buf), HAL_MAX_DELAY);
    HAL_Delay(300);
  }
  /* USER CODE END 3 */
