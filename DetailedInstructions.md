## Exercise 1 — Digital I/O

### What you are building
Three C modules:
1. `gpio` — generic pin configure/read/write
2. `led` — encapsulated 8-LED array driver (PE8–PE15)
3. `button` — interrupt-driven user button with callback (PA0)

---

### Step 1 — CubeMX / Project setup

1. Open STM32CubeIDE → New STM32 Project → select **STM32F303VCTx**
2. In **Pinout & Configuration**:
   - Set **PE8 – PE15** to `GPIO_Output`
   - Set **PA0** to `GPIO_EXTI0`
3. Under **System Core → GPIO → PA0**: set Mode to `External Interrupt Mode with Rising edge trigger`
4. Under **System Core → NVIC**: enable `EXTI line0 interrupt`
5. **Project Manager** → set project name (e.g. `Exercise1`) → Generate Code

### Step 3 — Create `led.h` and `led.c`

Create `Core/Inc/led.h`:
```c
#ifndef LED_H
#define LED_H

#include <stdint.h>

void    led_init(void);
void    led_set(uint8_t state);
uint8_t led_get(void);

#endif
```

Create `Core/Src/led.c`:
```c
#include "led.h"
#include "gpio.h"
#include "stm32f3xx_hal.h"

#define LED_PORT  GPIOE
#define LED_FIRST GPIO_PIN_8   // PE8 = bit 0 of state

static uint8_t led_state = 0;

void led_init(void) {
    __HAL_RCC_GPIOE_CLK_ENABLE();
    uint16_t all_pins = GPIO_PIN_8  | GPIO_PIN_9  | GPIO_PIN_10 | GPIO_PIN_11 |
                        GPIO_PIN_12 | GPIO_PIN_13 | GPIO_PIN_14 | GPIO_PIN_15;
    gpio_init(LED_PORT, all_pins, GPIO_MODE_OUTPUT_PP, GPIO_NOPULL);
}

void led_set(uint8_t state) {
    led_state = state;
    for (int i = 0; i < 8; i++) {
        GPIO_PinState val = (state & (1 << i)) ? GPIO_PIN_SET : GPIO_PIN_RESET;
        gpio_write(LED_PORT, (LED_FIRST << i), val);
    }
}

uint8_t led_get(void) {
    return led_state;
}
```

### Step 4 — Create `button.h` and `button.c`

Create `Core/Inc/button.h`:

Create `Core/Src/button.c`:
```c
#include "button.h"
#include "stm32f3xx_hal.h"

static ButtonCallback button_cb = NULL;

void button_init(ButtonCallback cb) {
    button_cb = cb;

    __HAL_RCC_GPIOA_CLK_ENABLE();

    GPIO_InitTypeDef cfg = {0};
    cfg.Pin  = GPIO_PIN_0;
    cfg.Mode = GPIO_MODE_IT_RISING;
    cfg.Pull = GPIO_NOPULL;
    HAL_GPIO_Init(GPIOA, &cfg);

    HAL_NVIC_SetPriority(EXTI0_IRQn, 2, 0);
    HAL_NVIC_EnableIRQ(EXTI0_IRQn);
}

/* IRQ handler — place in stm32f3xx_it.c if not already there */
void EXTI0_IRQHandler(void) {
    HAL_GPIO_EXTI_IRQHandler(GPIO_PIN_0);
}

/* HAL calls this after clearing the EXTI pending bit */
void HAL_GPIO_EXTI_Callback(uint16_t GPIO_Pin) {
    if (GPIO_Pin == GPIO_PIN_0 && button_cb != NULL) {
        button_cb();
    }
}
```

> **Note:** If `EXTI0_IRQHandler` is already defined in `stm32f3xx_it.c` (generated by CubeMX), add `HAL_GPIO_EXTI_IRQHandler(GPIO_PIN_0);` inside it instead of redefining it.

### Advanced (Task e) — Rate-limiting with a timer

Add a flag inside `led.c` and a timer interrupt that clears it:

```c
// Inside led.c
static volatile uint8_t led_ready = 1;   // 1 = allowed to change

void led_set(uint8_t state) {
    if (!led_ready) return;  // silently ignore if rate-limited
    led_ready = 0;
    led_state = state;
    for (int i = 0; i < 8; i++) {
        GPIO_PinState val = (state & (1 << i)) ? GPIO_PIN_SET : GPIO_PIN_RESET;
        gpio_write(LED_PORT, (LED_FIRST << i), val);
    }
    // TIM6 configured to fire after e.g. 200ms and set led_ready = 1
}

// Called from TIM6 period elapsed callback
void led_rate_timer_callback(void) {
    led_ready = 1;
}
```

Enable TIM6 in CubeMX (basic timer, no channel needed), period = 200 ms. In `stm32f3xx_it.c`, call `led_rate_timer_callback()` from `TIM6_DAC_IRQHandler`.

## Exercise 2 — Timer Interface / PWM

### What you are building
1. `timer` — repeating hardware timer with callback, get/set period, one-shot mode
2. `servo` — PWM output using the timer module to position a hobby servo

---

### Step 1 — CubeMX setup

1. Under **Timers → TIM2**:
   - Clock Source: `Internal Clock`
   - Prescaler: `7999` (gives 1 ms tick at 8 MHz, or adjust for your clock)
   - Counter Period: `19` (20 ticks × 1 ms = 20 ms = 50 Hz)
   - auto-reload preload: `Enable`
2. Under **NVIC**: enable `TIM2 global interrupt`
3. For the servo output GPIO, set the chosen pin (e.g. **PA6**) to `GPIO_Output`

> If using 72 MHz PLL clock, set Prescaler to `71999` for 1 ms ticks.

---

### Step 2 — Create `timer.h` and `timer.c`

Create `Core/Inc/timer.h`:
```c
#ifndef TIMER_H
#define TIMER_H

#include <stdint.h>

typedef void (*TimerCallback)(void);

void     timer_init(uint32_t period_ms, TimerCallback cb);
void     timer_set_period(uint32_t period_ms);
uint32_t timer_get_period(void);
void     timer_stop(void);
void     timer_oneshot(uint32_t delay_ms, TimerCallback cb);   // Advanced

#endif
```

Create `Core/Src/timer.c`:
```c
#include "timer.h"
#include "stm32f3xx_hal.h"

extern TIM_HandleTypeDef htim2;   // generated by CubeMX

static TimerCallback timer_cb    = NULL;
static uint32_t      timer_period = 20;
static uint8_t       oneshot_mode = 0;

void timer_init(uint32_t period_ms, TimerCallback cb) {
    timer_cb     = cb;
    timer_period = period_ms;
    __HAL_TIM_SET_AUTORELOAD(&htim2, period_ms - 1);
    HAL_TIM_Base_Start_IT(&htim2);
}

void timer_set_period(uint32_t period_ms) {
    timer_period = period_ms;
    __HAL_TIM_SET_AUTORELOAD(&htim2, period_ms - 1);
}

uint32_t timer_get_period(void) {
    return timer_period;
}

void timer_stop(void) {
    HAL_TIM_Base_Stop_IT(&htim2);
}

void timer_oneshot(uint32_t delay_ms, TimerCallback cb) {
    timer_cb    = cb;
    oneshot_mode = 1;
    __HAL_TIM_SET_AUTORELOAD(&htim2, delay_ms - 1);
    __HAL_TIM_SET_COUNTER(&htim2, 0);
    HAL_TIM_Base_Start_IT(&htim2);
}

/* Called by HAL when TIM2 period elapses */
void HAL_TIM_PeriodElapsedCallback(TIM_HandleTypeDef *htim) {
    if (htim->Instance == TIM2) {
        if (oneshot_mode) {
            HAL_TIM_Base_Stop_IT(&htim2);
            oneshot_mode = 0;
        }
        if (timer_cb != NULL) {
            timer_cb();
        }
    }
}
```

### Step 3 — Create `servo.h` and `servo.c`

Create `Core/Inc/servo.h`:
```c
#ifndef SERVO_H
#define SERVO_H

void  servo_init(void);
void  servo_set_position(float pos);   // 0.0 = CW (1ms), 0.5 = centre, 1.0 = CCW (2ms)

#endif
```

Create `Core/Src/servo.c`:

The servo PWM works by running a 50 Hz (20 ms) cycle. At the start of each cycle the output pin goes HIGH; after 1–2 ms (depending on position) it goes LOW. We implement this with two timer events per cycle using TIM3 in PWM mode or by toggling a pin manually from TIM2 callbacks.

**Simple two-interrupt approach using TIM2:**

```c
#include "servo.h"
#include "timer.h"
#include "gpio.h"
#include "stm32f3xx_hal.h"

// Servo output: PA6
#define SERVO_PORT  GPIOA
#define SERVO_PIN   GPIO_PIN_6

// Pulse width in ms (1.0 to 2.0)
static float  pulse_ms   = 1.5f;
static uint8_t pwm_phase = 0;     // 0 = high phase, 1 = low phase (fill remaining 18ms)

static void pwm_callback(void) {
    if (pwm_phase == 0) {
        // Rising edge: go HIGH, next interrupt after pulse_ms
        gpio_write(SERVO_PORT, SERVO_PIN, GPIO_PIN_SET);
        timer_set_period((uint32_t)(pulse_ms));   // 1 or 2 ms
        pwm_phase = 1;
    } else {
        // Falling edge: go LOW, next interrupt fills rest of 20ms period
        gpio_write(SERVO_PORT, SERVO_PIN, GPIO_PIN_RESET);
        timer_set_period((uint32_t)(20.0f - pulse_ms));
        pwm_phase = 0;
    }
}

void servo_init(void) {
    __HAL_RCC_GPIOA_CLK_ENABLE();
    gpio_init(SERVO_PORT, SERVO_PIN, GPIO_MODE_OUTPUT_PP, GPIO_NOPULL);
    gpio_write(SERVO_PORT, SERVO_PIN, GPIO_PIN_RESET);
    timer_init(1, pwm_callback);   // start with 1ms first phase
}

void servo_set_position(float pos) {
    if (pos < 0.0f) pos = 0.0f;
    if (pos > 1.0f) pos = 1.0f;
    pulse_ms = 1.0f + pos;   // maps 0→1ms, 0.5→1.5ms, 1.0→2ms
}
```

## Exercise 3 — Serial Interface

### What you are building
A UART module with:
- `serial_send_string()` — debug output to PC over USART1
- `serial_send_msg()` — structured framed packet with checksum
- Interrupt-driven receive with callback on terminating character

---

### Step 1 — CubeMX setup

1. **Connectivity → USART1**: Mode = `Asynchronous`, Baud = `115200`, PA9 (TX), PA10 (RX)
   - Used for PC debug console
2. **Connectivity → UART4**: Mode = `Asynchronous`, Baud = `115200`, PC10 (TX), PC11 (RX)
   - Used for board-to-board communication
3. Under **NVIC**: enable `USART1 global interrupt` and `UART4 global interrupt`

---

### Step 2 — Packet format

```
[ 0xAA | SIZE | TYPE | BODY (SIZE bytes) | CHECKSUM | 0x55 ]
  start  1 byte 1byte  n bytes             XOR of all  stop
```

Checksum = XOR of SIZE, TYPE, and all BODY bytes.

---

### Step 3 — Create `serial.h` and `serial.c`

Create `Core/Inc/serial.h`:
```c
#ifndef SERIAL_H
#define SERIAL_H

#include "stm32f3xx_hal.h"
#include <stdint.h>

#define SERIAL_BUFFER_SIZE  128
#define SERIAL_START_BYTE   0xAA
#define SERIAL_STOP_BYTE    0x55

typedef void (*SerialRxCallback)(uint8_t *buffer, uint16_t length);

void serial_init(UART_HandleTypeDef *uart, SerialRxCallback cb);
void serial_send_string(UART_HandleTypeDef *uart, const char *str);
void serial_send_msg(UART_HandleTypeDef *uart, uint8_t type, void *data, uint16_t size);

/* Call this from the UART IRQ handler */
void serial_irq_handler(UART_HandleTypeDef *uart);

#endif
```

Create `Core/Src/serial.c`:
```c
#include "serial.h"
#include <string.h>

static SerialRxCallback rx_callback = NULL;
static UART_HandleTypeDef *rx_uart  = NULL;

static uint8_t  rx_buf[SERIAL_BUFFER_SIZE];
static uint16_t rx_idx = 0;
static uint8_t  rx_byte;      // single byte HAL receive target

/* ---- Initialisation ---- */
void serial_init(UART_HandleTypeDef *uart, SerialRxCallback cb) {
    rx_uart    = uart;
    rx_callback = cb;
    rx_idx     = 0;
    // Kick off first single-byte interrupt receive
    HAL_UART_Receive_IT(uart, &rx_byte, 1);
}

/* ---- Debug string send ---- */
void serial_send_string(UART_HandleTypeDef *uart, const char *str) {
    HAL_UART_Transmit(uart, (uint8_t *)str, strlen(str), HAL_MAX_DELAY);
}

/* ---- Structured message send ---- */
void serial_send_msg(UART_HandleTypeDef *uart, uint8_t type, void *data, uint16_t size) {
    uint8_t pkt[SERIAL_BUFFER_SIZE];
    uint16_t idx = 0;
    uint8_t checksum = 0;

    pkt[idx++] = SERIAL_START_BYTE;
    pkt[idx++] = (uint8_t)size;
    pkt[idx++] = type;

    checksum ^= (uint8_t)size;
    checksum ^= type;

    uint8_t *body = (uint8_t *)data;
    for (uint16_t i = 0; i < size; i++) {
        pkt[idx++] = body[i];
        checksum  ^= body[i];
    }

    pkt[idx++] = checksum;
    pkt[idx++] = SERIAL_STOP_BYTE;

    HAL_UART_Transmit(uart, pkt, idx, HAL_MAX_DELAY);
}

/* ---- IRQ receive handler — call from HAL_UART_RxCpltCallback ---- */
void serial_irq_handler(UART_HandleTypeDef *uart) {
    if (uart->Instance != rx_uart->Instance) return;

    if (rx_idx < SERIAL_BUFFER_SIZE) {
        rx_buf[rx_idx++] = rx_byte;
    }

    // Terminating character received
    if (rx_byte == SERIAL_STOP_BYTE && rx_idx >= 5) {
        // Validate checksum
        uint8_t expected = 0;
        for (uint16_t i = 1; i < rx_idx - 2; i++) {   // SIZE to end of body
            expected ^= rx_buf[i];
        }
        if (expected == rx_buf[rx_idx - 2] && rx_callback != NULL) {
            rx_callback(rx_buf, rx_idx);
        }
        rx_idx = 0;   // reset buffer
    }

    // Re-arm for next byte
    HAL_UART_Receive_IT(rx_uart, &rx_byte, 1);
}
```

Add to `stm32f3xx_it.c` (or `main.c`):
```c
void HAL_UART_RxCpltCallback(UART_HandleTypeDef *huart) {
    serial_irq_handler(huart);
}
```

## Exercise 4 — I2C Compass / Magnetometer

### What you are building
An I2C driver for the **LSM303DLHC** magnetometer on the Discovery Board. Reads X/Y/Z field values, decodes a heading, and stores results in a shared `CompassData` struct with a timestamp.

---

### Step 1 — CubeMX setup

1. **Connectivity → I2C1**: Mode = `I2C`, Speed = `100 kHz` (standard mode)
   - SCL = **PB6**, SDA = **PB7** (default for I2C1 on STM32F3 Discovery)
2. Leave everything else at default — no interrupts needed for basic polling

---

### Step 2 — LSM303DLHC register reference

| Register | Address | Purpose |
|---|---|---|
| `CRB_REG_M` | `0x01` | Gain setting (set to `0x20` for ±1.3 Gauss) |
| `MR_REG_M` | `0x02` | Mode: `0x00` = continuous measurement |
| `OUT_X_H_M` | `0x03` | X high byte (read 6 bytes: X_H, X_L, Z_H, Z_L, Y_H, Y_L) |

**7-bit I2C address of magnetometer:** `0x1E`

---

### Step 3 — Create `compass.h` and `compass.c`

Create `Core/Inc/compass.h`:
```c
#ifndef COMPASS_H
#define COMPASS_H

#include <stdint.h>

typedef struct {
    int16_t  raw_x;
    int16_t  raw_y;
    int16_t  raw_z;
    float    heading;     // degrees, 0.0 to 360.0
    uint32_t timestamp;   // HAL_GetTick() at time of read
} CompassData;

void         compass_init(void);
void         compass_read(void);
CompassData *compass_get_data(void);

#endif
```

Create `Core/Src/compass.c`:
```c
#include "compass.h"
#include "stm32f3xx_hal.h"
#include <math.h>

#define MAG_ADDR        (0x1E << 1)   // HAL uses 8-bit address (7-bit << 1)
#define MAG_CRB_REG     0x01
#define MAG_MR_REG      0x02
#define MAG_OUT_X_H     0x03

extern I2C_HandleTypeDef hi2c1;

static CompassData data = {0};

static void mag_write_reg(uint8_t reg, uint8_t val) {
    uint8_t buf[2] = { reg, val };
    HAL_I2C_Master_Transmit(&hi2c1, MAG_ADDR, buf, 2, HAL_MAX_DELAY);
}

void compass_init(void) {
    HAL_Delay(10);   // wait for sensor power-up
    mag_write_reg(MAG_CRB_REG, 0x20);   // ±1.3 Gauss gain
    mag_write_reg(MAG_MR_REG,  0x00);   // continuous measurement mode
}

void compass_read(void) {
    uint8_t reg = MAG_OUT_X_H;
    uint8_t raw[6];

    // Point to output register, then read 6 bytes
    HAL_I2C_Master_Transmit(&hi2c1, MAG_ADDR, &reg, 1, HAL_MAX_DELAY);
    HAL_I2C_Master_Receive(&hi2c1,  MAG_ADDR, raw, 6, HAL_MAX_DELAY);

    // Register order: X_H, X_L, Z_H, Z_L, Y_H, Y_L
    data.raw_x = (int16_t)((raw[0] << 8) | raw[1]);
    data.raw_z = (int16_t)((raw[2] << 8) | raw[3]);
    data.raw_y = (int16_t)((raw[4] << 8) | raw[5]);

    // Decode heading (tilt-compensated heading requires accelerometer — basic 2D version)
    float heading_rad = atan2f((float)data.raw_y, (float)data.raw_x);
    float heading_deg = heading_rad * (180.0f / (float)M_PI);
    if (heading_deg < 0.0f) heading_deg += 360.0f;

    data.heading   = heading_deg;
    data.timestamp = HAL_GetTick();
}

CompassData *compass_get_data(void) {
    return &data;
}
```

> **Important:** Add `-lm` to linker flags if `math.h` is not linking: Project → Properties → C/C++ Build → Settings → MCU GCC Linker → Libraries → add `m`.

---

## Exercise 5 — Integration Task

### What you are building
Two communicating boards:
- **Board 1 (sender):** reads compass + button → sends UART packets
- **Board 2 (receiver):** receives packets → drives servo or LED display

---

### Step 1 — Hardware connections

```
Board 1 PC10 (TX)  ──────────────→  Board 2 PC11 (RX)
Board 1 GND        ─────────────── Board 2 GND
Board 2 PA6 (PWM)  → Servo signal wire
```

---

### Step 2 — Define the message structure

Create `Core/Inc/messages.h` (use on **both** boards):
```c
#ifndef MESSAGES_H
#define MESSAGES_H

#include <stdint.h>

#define MSG_TYPE_COMPASS  0x01

typedef struct {
    float    heading;        // 0.0 – 360.0 degrees
    int16_t  raw_x;
    int16_t  raw_y;
    int16_t  raw_z;
    uint32_t timestamp;
    uint8_t  display_mode;  // 0 = servo, 1 = LED
} CompassMsg;

#endif
```

---

### Step 3 — Board 1 firmware

In `main.c`:
```c
#include "compass.h"
#include "button.h"
#include "serial.h"
#include "messages.h"

extern UART_HandleTypeDef huart1;   // debug
extern UART_HandleTypeDef huart4;   // to Board 2

static uint8_t display_mode = 0;

static void on_button(void) {
    display_mode ^= 1;   // toggle between servo and LED mode
}

int main(void) {
    HAL_Init();
    SystemClock_Config();
    MX_I2C1_Init();
    MX_USART1_UART_Init();
    MX_UART4_Init();

    compass_init();
    button_init(on_button);
    serial_init(&huart4, NULL);   // no receive callback needed on Board 1

    serial_send_string(&huart1, "Board 1 ready\r\n");

    while (1) {
        compass_read();
        CompassData *d = compass_get_data();

        CompassMsg msg = {
            .heading      = d->heading,
            .raw_x        = d->raw_x,
            .raw_y        = d->raw_y,
            .raw_z        = d->raw_z,
            .timestamp    = d->timestamp,
            .display_mode = display_mode,
        };

        serial_send_msg(&huart4, MSG_TYPE_COMPASS, &msg, sizeof(msg));

        char buf[64];
        sprintf(buf, "Sent heading: %.1f  mode: %d\r\n", msg.heading, display_mode);
        serial_send_string(&huart1, buf);

        HAL_Delay(100);
    }
}
```

### Step 4 — Board 2 firmware

In `main.c`:
```c
#include "led.h"
#include "servo.h"
#include "serial.h"
#include "messages.h"
#include <string.h>

extern UART_HandleTypeDef huart1;   // debug
extern UART_HandleTypeDef huart4;   // from Board 1

static void on_receive(uint8_t *buf, uint16_t len) {
    // buf = full raw packet: [0xAA, size, type, body..., checksum, 0x55]
    if (buf[2] != MSG_TYPE_COMPASS) return;

    CompassMsg msg;
    memcpy(&msg, &buf[3], sizeof(CompassMsg));

    if (msg.display_mode == 0) {
        // Servo mode: map heading 0-360 to position 0.0-1.0
        servo_set_position(msg.heading / 360.0f);
        led_set(0x00);   // LEDs off in servo mode
    } else {
        // LED mode: light one of 8 LEDs for the heading octant
        servo_set_position(0.5f);   // centre servo
        uint8_t octant = (uint8_t)((msg.heading + 22.5f) / 45.0f) % 8;
        led_set(1 << octant);
    }
}

int main(void) {
    HAL_Init();
    SystemClock_Config();
    MX_USART1_UART_Init();
    MX_UART4_Init();
    MX_TIM2_Init();

    led_init();
    servo_init();
    serial_init(&huart4, on_receive);

    serial_send_string(&huart1, "Board 2 ready\r\n");

    while (1) {}   // everything is interrupt-driven
}
```

### Step 6 — How to verify it works end to end

| Test | What to do | Expected result |
|---|---|---|
| Basic comms | Open USART1 terminal on both boards | Board 1 prints heading, Board 2 receives silently |
| Servo tracking | Rotate Board 1 | Servo on Board 2 moves in proportion to heading |
| Full CW | Point Board 1 to heading ~0° | Servo at ~1 ms (full clockwise position) |
| Full CCW | Point Board 1 to heading ~360° | Servo at ~2 ms (full counterclockwise) |
| Button toggle | Press button on Board 1 | Board 2 switches from servo movement to LED display |
| LED North | With LED mode active, point to North | PE8 lights up (octant 0) |
| LED East | Point to East (~90°) | PE10 lights up (octant 2) |
| Disconnect UART | Unplug wire between boards | Board 2 holds last position — no crash |

## General Debugging Tips

- Add `serial_send_string(&huart1, "checkpoint\r\n")` at key points to trace execution
- Use the STM32CubeIDE **Live Expressions** panel to watch variables in real time while debugging
- If a module hangs, check for infinite loops in interrupt callbacks
- If I2C reads all `0xFF`, the sensor is not ACKing — check address and wiring
- If the servo jitters, the PWM period or pulse width math is off — verify on an oscilloscope
